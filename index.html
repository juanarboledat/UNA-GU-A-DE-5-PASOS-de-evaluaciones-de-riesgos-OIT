            {

                question: "Según la guía, un 'peligro' se define como:",

                correctAnswer: "Cualquier cosa que pueda ocasionar un daño (productos químicos, electricidad, etc.).",

                incorrectAnswer: "La probabilidad estadística de que una empresa sea multada por inspección laboral.",

                explanation: "Un peligro es cualquier cosa que pueda ocasionar daño, como productos químicos, electricidad, etc."

            },

            {

                question: "La guía menciona que mejorar la seguridad y la salud:",

                correctAnswer: "No implica necesariamente costos elevados; hay precauciones de bajo coste muy efectivas.",

                incorrectAnswer: "Siempre requiere una inversión económica grande en tecnología de última generación.",

                explanation: "Mejorar la seguridad no siempre implica costos elevados; existen precauciones de bajo coste muy efectivas."

            }

        ];


        // Variables del juego

        let currentQuestion = 0;

        let score = 0;

        let selectedOption = null;


        // Elementos DOM

        const screens = document.querySelectorAll('.screen');

        const startBtn = document.querySelector('.btn-start');

        const nextBtn = document.getElementById('next-btn');

        const restartBtn = document.querySelector('.btn-restart');

        const questionElement = document.getElementById('question');

        const optionsElement = document.getElementById('options');

        const feedbackElement = document.getElementById('feedback');

        const scoreElement = document.getElementById('score');

        const finalScoreElement = document.getElementById('final-score');

        const questionNumberElement = document.getElementById('question-number');

        const progressBar = document.getElementById('progress-bar');


        // Inicializar el juego

        startBtn.addEventListener('click', startGame);

        nextBtn.addEventListener('click', nextQuestion);

        restartBtn.addEventListener('click', restartGame);


        // Iniciar el juego

        function startGame() {

            showScreen('game-screen');

            loadQuestion();

        }


        // Mostrar pantalla específica

        function showScreen(screenClass) {

            screens.forEach(screen => {

                screen.classList.remove('active');

            });

            document.querySelector(`.${screenClass}`).classList.add('active');

        }


        // Cargar pregunta actual

        function loadQuestion() {

            const question = questions[currentQuestion];

            questionElement.textContent = question.question;

            questionNumberElement.textContent = currentQuestion + 1;

            

            // Mezclar las respuestas

            const answers = [question.correctAnswer, question.incorrectAnswer];

            if (Math.random() > 0.5) answers.reverse();

            

            optionsElement.innerHTML = '';

            answers.forEach(answer => {

                const option = document.createElement('div');

                option.classList.add('option');

                option.textContent = answer;

                option.addEventListener('click', () => selectOption(option, answer === question.correctAnswer));

                optionsElement.appendChild(option);

            });

            

            // Ocultar feedback y botón siguiente

            feedbackElement.style.display = 'none';

            nextBtn.style.display = 'none';

            

            // Actualizar barra de progreso

            progressBar.style.width = `${(currentQuestion / questions.length) * 100}%`;

        }


        // Seleccionar opción

        function selectOption(optionElement, isCorrect) {

            if (selectedOption) return; // Evitar múltiples selecciones

            

            selectedOption = optionElement;

            const options = document.querySelectorAll('.option');

            options.forEach(option => {

                option.removeEventListener('click', () => {});

                option.style.cursor = 'default';

            });

            

            if (isCorrect) {

                optionElement.classList.add('correct');

                score++;

                scoreElement.textContent = score;

                showFeedback(true, questions[currentQuestion].explanation);

            } else {

                optionElement.classList.add('incorrect');

                // Marcar la respuesta correcta

                options.forEach(option => {

                    if (option.textContent === questions[currentQuestion].correctAnswer) {

                        option.classList.add('correct');

                    }

                });

                showFeedback(false, questions[currentQuestion].explanation);

            }

            

            nextBtn.style.display = 'block';

        }


        // Mostrar retroalimentación

        function showFeedback(isCorrect, explanation) {

            feedbackElement.textContent = explanation;

            feedbackElement.className = 'feedback ' + (isCorrect ? 'correct' : 'incorrect');

            feedbackElement.style.display = 'block';

        }


        // Siguiente pregunta

        function nextQuestion() {

            currentQuestion++;

            selectedOption = null;

            

            if (currentQuestion < questions.length) {

                loadQuestion();

            } else {

                endGame();

            }

        }


        // Terminar el juego

        function endGame() {

            finalScoreElement.textContent = `${score}/${questions.length}`;

            showScreen('completion-screen');

        }


        // Reiniciar el juego

        function restartGame() {

            currentQuestion = 0;

            score = 0;

            selectedOption = null;

            scoreElement.textContent = score;

            showScreen('welcome-screen');

        }
